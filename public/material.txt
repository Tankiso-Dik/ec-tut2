INDEX BY records must practice







SET SERVEROUTPUT ON

DECLARE
TYPE is_subj_table_type IS TABLE OF
s_subject%ROWTYPE INDEX BY BINARY_INTEGER;

is_subj_table is_subj_table_type;

BEGIN
FOR i IN 1..3 LOOP
SELECT *
INTO is_subj_table(i)
FROM s_subject
WHERE SUBSTR(subj_code,1,2) = 'IS'
AND SUBSTR(subj_code,-1) = i;
END LOOP;

DBMS_OUTPUT.PUT_LINE('INFORMATION SYSTEMS SUBJECT CLUSTER');
DBMS_OUTPUT.PUT_LINE('===================================');

FOR i IN is_subj_table.FIRST..is_subj_table.LAST LOOP
DBMS_OUTPUT.PUT_LINE(is_subj_table(i).subj_code || ' - ' || is_subj_table(i).subj_name);
END LOOP;

END;
/









Explicit Cursor Using ROWTYPE

SET SERVEROUTPUT ON

DECLARE

CURSOR fac_cur IS 
SELECT *
FROM s_faculty;

v_fac fac_cur%ROWTYPE;

BEGIN

IF NOT fac_cur%ISOPEN THEN 
OPEN fac_cur;
END IF;

LOOP
FETCH fac_cur INTO v_fac;

IF fac_cur%FOUND THEN
DBMS_OUTPUT.PUT_LINE(v_fac.fac_code || ' - ' || v_fac.fac_name);
END IF;

EXIT WHEN fac_cur%NOTFOUND;

END LOOP;

DBMS_OUTPUT.PUT_LINE('Total number of rows: ' || fac_cur%ROWCOUNT);

CLOSE fac_cur;

END;
/





Explicit Cursor WHILE LOOP



SET SERVEROUTPUT ON

DECLARE

CURSOR dip_cur IS 
SELECT *
FROM s_diploma;

v_dip dip_cur%ROWTYPE;

BEGIN

IF NOT dip_cur%ISOPEN THEN
OPEN dip_cur;
END IF;

FETCH dip_cur INTO v_dip;

WHILE dip_cur%FOUND LOOP

DBMS_OUTPUT.PUT_LINE(v_dip.dip_code || ' - ' || v_dip.dip_name);

FETCH dip_cur INTO v_dip;

END LOOP;

DBMS_OUTPUT.PUT_LINE('Total Number of rows: ' || dip_cur%ROWCOUNT);

CLOSE dip_cur;

END;
/






CURSOR FOR LOOP NESTED



SET SERVEROUTPUT ON

DECLARE

CURSOR fac_cur IS 
SELECT * 
FROM s_faculty;

CURSOR dip_cur(p_fac_code s_diploma.fac_code%TYPE) IS
SELECT *
FROM s_diploma
WHERE fac_code = p_fac_code;

BEGIN

FOR v_fac IN fac_cur LOOP

DBMS_OUTPUT.PUT_LINE('FACULTY: ' || v_fac.fac_name);
DBMS_OUTPUT.PUT_LINE('---------------------------------------------------');

FOR v_dip IN dip_cur(v_fac.fac_code) LOOP

DBMS_OUTPUT.PUT_LINE(' ' || v_dip.dip_code || ' - ' || v_dip.dip_name);

END LOOP;

END LOOP;

END;
/





OUTPUT CONTROL

SET SERVEROUTPUT ON;

DECLARE

   CURSOR test_cur IS
      SELECT f.fac_code, f.fac_name, d.dip_code, d.dip_name
      FROM s_faculty f, s_diploma d
      WHERE f.fac_code = d.fac_code
      ORDER BY f.fac_code;

   v_rec test_cur%ROWTYPE;
   v_prev_fac_code s_faculty.fac_code%TYPE := NULL;

BEGIN

   OPEN test_cur;

   LOOP
      FETCH test_cur INTO v_rec;
      EXIT WHEN test_cur%NOTFOUND;

      IF v_prev_fac_code IS NULL OR v_prev_fac_code != v_rec.fac_code THEN
         DBMS_OUTPUT.PUT_LINE('FACULTY: ' || v_rec.fac_name);
      END IF;

      DBMS_OUTPUT.PUT_LINE('   ' || v_rec.dip_code);

      v_prev_fac_code := v_rec.fac_code;

   END LOOP;

   CLOSE test_cur;

END;
/




Exception Block

SET SERVEROUTPUT ON;

DECLARE

   v_author_id s_author.author_id%TYPE := '&author_id';
   v_book_name s_book.book_name%TYPE;

BEGIN

   SELECT book_name
   INTO v_book_name
   FROM s_book
   WHERE author_id = v_author_id;

   DBMS_OUTPUT.PUT_LINE(v_book_name);

EXCEPTION

   WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE(v_author_id || ' does not exist');

   WHEN TOO_MANY_ROWS THEN
      DBMS_OUTPUT.PUT_LINE('Many books under ' || v_author_id);

END;
/

SET SERVEROUTPUT ON;

DECLARE

   v_regno vehicle.regno%TYPE := '&enter_registration_number';
   v_vehicle vehicle%ROWTYPE;

BEGIN

   SELECT *
   INTO v_vehicle
   FROM vehicle
   WHERE regno = v_regno;

   DBMS_OUTPUT.PUT_LINE('--- VEHICLE AUDIT REPORT ---');

   DBMS_OUTPUT.PUT_LINE('MAKE: ' || UPPER(v_vehicle.make));

   DBMS_OUTPUT.PUT_LINE('REFERENCE: ' || v_vehicle.model || 
   ' (' || SUBSTR(v_vehicle.model,1,3) || ')');

   DBMS_OUTPUT.PUT_LINE('COLOUR: ' || NVL(v_vehicle.colour, 'Not Specified'));

   DBMS_OUTPUT.PUT_LINE('MASKED REG: ' || 
   SUBSTR(v_vehicle.regno,1,4) || RPAD('*',6,'*'));

   DBMS_OUTPUT.PUT_LINE('----------------------------');

EXCEPTION

   WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('Error: No vehicle found with registration: ' || v_regno);

END;

Question 2


SET SERVEROUTPUT ON;

DECLARE

   v_fuel_type vehicle.fueltype%TYPE := 'D';

   CURSOR c_diesel_audit IS
      SELECT v.make, v.model, v.regno, m.surname
      FROM vehicle v, jobcard j, mechanic m
      WHERE v.regno = j.regno
      AND j.mno = m.mno
      AND v.fueltype = v_fuel_type;

   v_rec c_diesel_audit%ROWTYPE;

   v_count NUMBER := 0;

BEGIN

   OPEN c_diesel_audit;

   FETCH c_diesel_audit INTO v_rec;

   WHILE c_diesel_audit%FOUND LOOP

      DBMS_OUTPUT.PUT_LINE(
         v_rec.make || ' ' || v_rec.model || ' | ' ||
         UPPER(v_rec.regno) || ' | ' ||
         INITCAP(v_rec.surname)
      );

      v_count := v_count + 1;

      FETCH c_diesel_audit INTO v_rec;

   END LOOP;

   IF v_count = 0 THEN
      DBMS_OUTPUT.PUT_LINE('Audit Complete: No diesel vehicles currently registered in the workshop logs.');
   END IF;

   CLOSE c_diesel_audit;

END;
/
